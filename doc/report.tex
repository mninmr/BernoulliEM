    \section{Bernoulli Distribution} 
    
\textbf{Observed data}: $X_1, X_2, \dots, X_N \; \sim \;$i.i.d. Bernoulli$(X\;|\;\mu)$. Each $X_i$ is a vector of 0's and 1's.
\begin{align*}
X_i = [x_{i1}, x_{i2}, \dots, x_{iD}]\quad x_{id} = \{0, 1\}.
\end{align*}

So, $N$ observations and $D$ features.
$$
X =
 \begin{pmatrix}
  x_{11} & x_{12} & \cdots & x_{1D} \\
  x_{21} & x_{22} & \cdots & x_{2D} \\
  \vdots  & \vdots  & \ddots & \vdots  \\
  x_{N1} & x_{N2} & \cdots & x_{ND}
 \end{pmatrix}.
 $$
\textbf{Density}:
\begin{align*}
\text{Bernoulli}(X_i\;|\;\mu) =\prod\limits_{d=1}^D \mu_d^{x_{id}}(1-\mu_d)^{1-x_{id}}.
\end{align*} 
Now, let's consider a finite mixture of $K$ Bernoulli distributions. Then, its density function can be expressed as follows:
\begin{align*}
\mathbb{P}(X_i\;|\;\mu, \pi) = \sum\limits_{k=1}^K \pi_k \mathbb{P}(X_i\;|\;\mu_k),
\end{align*} where $\mu = [\mu_1, \dots, \mu_K], \pi=[\pi_1, \dots, \pi_K]$, and $\mathbb{P}(X_i\; |\; \mu_k) = \prod_{d=1}^D \mu_{kd}^{x_{id}}(1-\mu_{kd})^{1-x_{id}}$.

    \section{Inference with EM}

Given the observed data $X = [X_1, X_2, \dots, X_N]$, we can write the log-likelihood function for this model as follows:
\begin{align*}
\log \mathbb{P}(X\; |\; \mu, \pi) = \sum_{i=1}^N \log \left[\sum_{k=1}^K \pi_k \mathbb{P}(X_i\; |\; \mu_k)\right]. 
\end{align*}    
We cannot solve it for the maximum likelihood solution in closed form due to the summation within the logarithm. We now derive the EM algorithm for mixture of Bernoulli distributions. Let the latent variable $z_i = [z_{i1}, \dots, z_{iK}]$ be a binary $K$-dimensional variable with a single component equal to $1$ and the rest equal to $0$, indicating the cluster assignment of $X_i$. 

    Now, the expectation of the log-likelihood is given by:
    \begin{align*}
    \mathbb{E}[\log \mathbb{P}(X, Z\; |\; \mu, \pi) ]= \sum_{i=1}^N\sum_{k=1}^K r_{ik} \left[\log \pi_k + \sum_{d=1}^D x_{id}\log \mu_{kd} + (1-x_{id})\log(1-\mu_{kd})\right],
    \end{align*} where $r_{ik} = \mathbb{E}[z_{ik}]$ is the probability that $X_i$ belongs to cluster $k$. 
    
\vspace{.05in}
    \noindent\textbf{E-step:} Compute the probabilities of cluster assignments. 
    \begin{align*}
    r_{ik} &= \cfrac{\pi_k \mathbb{P}(X_i\; |\; \mu_k)}{\sum_{j=1}^K \pi_j \mathbb{P}(X_i\; |\; \mu_j)} \\
    &\propto \pi_k \prod_{d=1}^D\left[ \mu_{kd}^{x_{id}} (1-\mu_{kd})^{(1-x_{id})}\right]. 
    \end{align*}
    \noindent\textbf{M-step:} Update parameters $\mu$ and $\pi$ given $r$. 
    \begin{align*}
    \mu_k &= \frac{1}{N_k} \sum_{i=1}^N r_{ik} X_i \\
    \pi_k &= \frac{N_k}{N}, 
    \end{align*} where $N_k = \sum_{i=1}^N r_{ik}$.
